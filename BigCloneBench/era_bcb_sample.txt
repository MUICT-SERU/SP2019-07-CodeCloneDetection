select A.NAME,A.STARTLINE,A.ENDLINE,FUNCTIONS.NAME,FUNCTIONS.STARTLINE,FUNCTIONS.ENDLINE from
(select FUNCTIONS.NAME,FUNCTIONS.STARTLINE,FUNCTIONS.ENDLINE,CLONES.FUNCTION_ID_ONE,CLONES.FUNCTION_ID_TWO
from FUNCTIONS join CLONES on FUNCTIONS.ID=CLONES.FUNCTION_ID_ONE where functionality_id=4) as A
    join FUNCTIONS on A.FUNCTION_ID_TWO = FUNCTIONS.ID;


select A.NAME,A.STARTLINE,A.ENDLINE,FUNCTIONS.NAME,FUNCTIONS.STARTLINE,FUNCTIONS.ENDLINE,from
(select FUNCTIONS.NAME,FUNCTIONS.STARTLINE,FUNCTIONS.ENDLINE,false_positives.FUNCTION_ID_ONE,false_positives.FUNCTION_ID_TWO
from FUNCTIONS join false_positives on FUNCTIONS.ID=false_positives.FUNCTION_ID_ONE where functionality_id=4) as A
    join FUNCTIONS on A.FUNCTION_ID_TWO = FUNCTIONS.ID;
